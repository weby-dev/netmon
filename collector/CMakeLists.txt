cmake_minimum_required(VERSION 3.16)
project(netmon_collector CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

if(NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE Release)
endif()
add_compile_options(-O2 -Wall -Wextra -Wno-unused-parameter)

# ---- dependencies -------------------------------------------------------- #
find_package(PkgConfig REQUIRED)
pkg_check_modules(LIBBPF REQUIRED libbpf)
pkg_check_modules(LIBCURL REQUIRED libcurl)

find_package(Threads REQUIRED)

# libelf and zlib are needed by static libbpf linking.
find_library(LIBELF elf)
find_library(LIBZ z)

# ---- sources ------------------------------------------------------------- #
add_executable(netmon-collector
    src/main.cpp
    src/config.cpp
    src/net_util.cpp
    src/bpf_loader.cpp
    src/app_classifier.cpp
    src/l7_parser.cpp
    src/security_engine.cpp
    src/aggregator.cpp
    src/clickhouse_client.cpp
    src/stream_server.cpp
)

# Skeleton header + shared kernel/user header live here. The skeleton is
# generated by the top-level Makefile into collector/include/.
target_include_directories(netmon-collector PRIVATE
    ${CMAKE_CURRENT_SOURCE_DIR}/include
    ${CMAKE_CURRENT_SOURCE_DIR}/../ebpf       # common.h
    ${LIBBPF_INCLUDE_DIRS}
    ${LIBCURL_INCLUDE_DIRS}
)

target_link_libraries(netmon-collector PRIVATE
    ${LIBBPF_LIBRARIES}
    ${LIBCURL_LIBRARIES}
    ${LIBELF}
    ${LIBZ}
    Threads::Threads
)

target_link_directories(netmon-collector PRIVATE
    ${LIBBPF_LIBRARY_DIRS}
    ${LIBCURL_LIBRARY_DIRS}
)

install(TARGETS netmon-collector RUNTIME DESTINATION /usr/local/bin)
